PUSH_R macro REG_LIST
	IRP CUR_ITEM, <REG_LIST>
		IFIDN <CUR_ITEM>, <F>
			PUSHF
		ELSE
			PUSH CUR_ITEM
		ENDIF
	ENDM
ENDM

POP_R macro REG_LIST
	IRP CUR_ITEM, <REG_LIST>
		IFIDN <CUR_ITEM>, <F>
			POPF
		ELSE
			POP CUR_ITEM
		ENDIF
	ENDM
ENDM

CALL_FUNCTION macro FUNCTION_NAME:REQ, RETURN_VALUE:REQ, FUNCTION_PARAMS, REG_LIST

	local A, B, C
	A DB 0
	B DB 0
	C DB 8

	FUNCTION_NAME&:
	push EBP
	mov EBP, ESP

	IFNB <REG_LIST>
		IRP CUR_ITEM, <REG_LIST>
			PUSH CUR_ITEM
			@CatStr(QQ%A) TEXTEQU CUR_ITEM
			A = A + 1
		ENDM
	ELSE
		PUSH_R <EBX, ESI, EDI>
	ENDIF

	B = A

	C = 8
	IFNB <FUNCTION_PARAMS>
		IRP cur_p, <FUNCTION_PARAMS>
			cur_p TEXTEQU [EBP + C]
			C = C + 4
		ENDM
	ENDIF

	
	IFIDN <RETURN_VALUE>, <value>
		;вызываемая функция должна сохранять в регистр EAX возвращаемое числовое значение
	ENDIF
	IFIDN <RETURN_VALUE>, <pointer>
		;вызываемая функция должна сохранять в регистр EAX указатель
	ENDIF

	IFNB <REG_LIST>
		REPT A
			pop @CatStr(QQ&B)
			B = B - 1
		ENDM
	ELSE
		POP_R <EDI, ESI, EBX>
	ENDIF
	pop EBP
ENDM
